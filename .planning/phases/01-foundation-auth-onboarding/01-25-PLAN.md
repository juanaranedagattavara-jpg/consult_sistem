  ```typescript
  "use client";

  import { useForm } from "react-hook-form";
  import { useMutation, useQuery } from "convex/react";
  import { api } from "../../../convex/_generated/api";
  import { Button } from "@/components/ui/button";
  import { Input } from "@/components/ui/input";
  import { Label } from "@/components/ui/label";
  import { Card, CardContent, CardHeader, CardTitle } from "@/components/ui/card";
  import { toast } from "sonner";

  interface TimingData {
    confirmationHours: number;
    reminderHours: number;
    retries: number;
    waitlistTimeout: number;
  }

  export function TimingForm() {
    const user = useQuery(api.users.getCurrentUser);
    const updateTiming = useMutation(api.clinics.updateClinicTiming);

    const form = useForm<TimingData>({
      defaultValues: {
        confirmationHours: user?.clinic?.timing?.confirmationHours || 24,
        reminderHours: user?.clinic?.timing?.reminderHours || 2,
        retries: user?.clinic?.timing?.retries || 3,
        waitlistTimeout: user?.clinic?.timing?.waitlistTimeout || 30,
      },
    });

    const onSubmit = async (data: TimingData) => {
      try {
        await updateTiming({
          clinicId: user!.clinicId,
          timing: data,
        });
        toast.success("Configuración de tiempos actualizada");
      } catch (error: any) {
        toast.error(error.message || "Error al guardar");
      }
    };

    if (!user) return <div>Cargando...</div>;

    return (
      <Card>
        <CardHeader>
          <CardTitle>Tiempos y notificaciones</CardTitle>
        </CardHeader>
        <CardContent>
          <form onSubmit={form.handleSubmit(onSubmit)} className="space-y-4">
            <div>
              <Label htmlFor="confirmationHours">Confirmación (horas antes)</Label>
              <Input
                {...form.register("confirmationHours", { valueAsNumber: true })}
                id="confirmationHours"
                type="number"
                min={1}
                max={72}
                className="mt-2"
              />
              <p className="text-sm text-gray-500 mt-1">
                Cuántas horas antes de la cita se envía la confirmación
              </p>
            </div>

            <div>
              <Label htmlFor="reminderHours">Recordatorio (horas antes)</Label>
              <Input
                {...form.register("reminderHours", { valueAsNumber: true })}
                id="reminderHours"
                type="number"
                min={1}
                max={24}
                className="mt-2"
              />
              <p className="text-sm text-gray-500 mt-1">
                Cuántas horas antes se envía el recordatorio
              </p>
            </div>

            <div>
              <Label htmlFor="retries">Reintentos de WhatsApp</Label>
              <Input
                {...form.register("retries", { valueAsNumber: true })}
                id="retries"
                type="number"
                min={1}
                max={5}
                className="mt-2"
              />
              <p className="text-sm text-gray-500 mt-1">
                Intentos de contacto si no hay respuesta
              </p>
            </div>

            <div>
              <Label htmlFor="waitlistTimeout">Timeout lista de espera (minutos)</Label>
              <Input
                {...form.register("waitlistTimeout", { valueAsNumber: true })}
                id="waitlistTimeout"
                type="number"
                min={5}
                max={120}
                className="mt-2"
              />
              <p className="text-sm text-gray-500 mt-1">
                Tiempo para responder antes de pasar al siguiente
              </p>
            </div>

            <Button type="submit" disabled={form.formState.isSubmitting}>
              {form.formState.isSubmitting ? "Guardando..." : "Guardar"}
            </Button>
          </form>
        </CardContent>
      </Card>
    );
  }
  ```
