```tsx
"use client";

import * as React from "react";
import { Check } from "lucide-react";

import { cn } from "@/lib/utils";

// Preset colors for calendar display - distinguishable and accessible
const PRESET_COLORS = [
  { name: "Azul", value: "#2563eb" },
  { name: "Verde", value: "#16a34a" },
  { name: "Rojo", value: "#dc2626" },
  { name: "Amarillo", value: "#ca8a04" },
  { name: "Morado", value: "#9333ea" },
  { name: "Rosa", value: "#db2777" },
  { name: "Naranja", value: "#ea580c" },
  { name: "Cyan", value: "#0891b2" },
  { name: "Indigo", value: "#4f46e5" },
  { name: "Esmeralda", value: "#059669" },
  { name: "Slate", value: "#475569" },
  { name: "Zinc", value: "#71717a" },
];

interface ColorPickerProps {
  value: string;
  onChange: (value: string) => void;
  className?: string;
}

export function ColorPicker({ value, onChange, className }: ColorPickerProps) {
  return (
    <div className={cn("flex flex-wrap gap-2", className)}>
      {PRESET_COLORS.map((color) => (
        <button
          key={color.value}
          type="button"
          title={color.name}
          className={cn(
            "w-8 h-8 rounded-full border-2 flex items-center justify-center transition-all",
            value === color.value
              ? "border-foreground scale-110"
              : "border-transparent hover:scale-105"
          )}
          style={{ backgroundColor: color.value }}
          onClick={() => onChange(color.value)}
        >
          {value === color.value && (
            <Check className="h-4 w-4 text-white drop-shadow-md" />
          )}
        </button>
      ))}
    </div>
  );
}

// Export preset colors for use elsewhere if needed
export { PRESET_COLORS };
```

Key features:
- 12 preset colors that are visually distinct
- Simple swatch selection (not full spectrum)
- Shows checkmark on selected color
- Scale animation on hover/selected
- Accessible with title attribute for color names
- Exports PRESET_COLORS for reuse (e.g., random assignment)
